# project-structure
└── db.sqlite
└── database.js
├── controllers
│   └── categoriesController.js
│   └── volumesController.js
├── models
│   └── categoriesModel.js
│   └── categoriesClosureModel.js
│   └── volumesModel.js
├── serializers
│   └── categoriesSerializer.js
│   └── volumesSerializer.js
├── swagger_output.json
├── index.js (this will run the app and use swagger_output.json for the docs)
├── swagger.js (this will generate package_output.json)

# clone the project
-> git clone https://github.com/NlyArivony/digital_value.git

# install dependencies in package.json
-> npm install

# run app
-> node index.js

# Swagger doc to test the API's endpoints is available at:
http://localhost:3000/docs/
